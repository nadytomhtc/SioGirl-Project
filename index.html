<div id="ui-panel">

    <h3>SioGirl VRM 測試區</h3>

    <p>狀態：<span id="status">載入中...</span></p>

    <div id="chat-log" style="height: 300px; overflow-y: auto; background: #0f172a; padding: 10px; margin-bottom: 10px;"></div>

    <input id="user-input" type="text" placeholder="輸入訊息..." style="width: 70%; padding: 10px;">

    <button onclick="sayHello()">發送</button>

</div>

 

<script>

    // 1. 初始化 3D 視窗

    const container = document.getElementById('vrm-container');

    const scene = new THREE.Scene();

    const camera = new THREE.PerspectiveCamera(35, container.clientWidth / container.clientHeight, 0.1, 20);

    camera.position.set(0, 1.4, 1.2); // 這是聚焦在角色臉部的高度

 

    const renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true });

    renderer.setSize(container.clientWidth, container.clientHeight);

    container.appendChild(renderer.domElement);

 

    const light = new THREE.DirectionalLight(0xffffff);

    light.position.set(1, 1, 1).normalize();

    scene.add(light);

    scene.add(new THREE.AmbientLight(0xffffff, 0.4));

 

    // 2. 載入你的 VRM 檔案

    const loader = new THREE.GLTFLoader();

    // *** 請在這裡貼上你的 VRM 網址 ***

    const vrmUrl = 'https://raw.githubusercontent.com/pixiv/three-vrm/master/packages/three-vrm/examples/models/vrm/three-vrm-girl.vrm';

 

    loader.load(vrmUrl, (gltf) => {

        THREE.VRM.from(gltf).then((vrm) => {

            scene.add(vrm.scene);

            vrm.scene.rotation.y = Math.PI; // 轉向正面

            document.getElementById('status').innerText = "已就緒";

        });

    });

 

    function animate() {

        requestAnimationFrame(animate);

        renderer.render(scene, camera);

    }

    animate();

 

    // 3. 互動功能 (TTS 與 簡易回覆)

    function sayHello() {

        const input = document.getElementById('user-input');

        const text = input.value;

        if(!text) return;

       

        document.getElementById('chat-log').innerHTML += `<p>你: ${text}</p>`;

       

        const resp = "你好！我是 SioGirl，看到我的 3D 樣子了嗎？";

        document.getElementById('chat-log').innerHTML += `<p style="color:#60a5fa">SioGirl: ${resp}</p>`;

       

        const utter = new SpeechSynthesisUtterance(resp);

        utter.lang = 'zh-TW';

        window.speechSynthesis.speak(utter);

        input.value = "";

    }

</script>
